project(gfc3d_ipm CXX C)
cmake_minimum_required(VERSION 3.7)
cmake_policy(SET CMP0074 NEW)
set(CMAKE_BUILD_TYPE Release)
set(CMAKE_SKIP_BUILD_RPATH  FALSE)
set(CMAKE_BUILD_WITH_INSTALL_RPATH FALSE)
set(CMAKE_INSTALL_RPATH_USE_LINK_PATH TRUE)
set(siconos_DIR /usr/local/lib/cmake/siconos-4.5.0)
find_package(siconos 4.5.0 CONFIG REQUIRED)
set(CMAKE_MODULE_PATH ${siconos_ROOT}/lib/cmake/siconos-4.5.0)
if(ON)
set(Boost_NO_BOOST_CMAKE 1)
find_package(Boost 1.61 COMPONENTS )
endif()
add_executable(gfc3d_ipm /Users/minhnguyen/siconos/numerics/src/FrictionContact/gfc3d_ipm.c)
target_include_directories(gfc3d_ipm PUBLIC /Users/minhnguyen/siconos/numerics/src/FrictionContact)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::externals)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::numerics)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::kernel)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::control)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::mechanics)
target_link_libraries(gfc3d_ipm PRIVATE Siconos::io)
install(TARGETS gfc3d_ipm DESTINATION /Users/minhnguyen/siconos/numerics/src/FrictionContact)
